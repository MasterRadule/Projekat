	putanjaDatoteke = getPath();
	resetujModel();
	snimiModel();

SAVE - if(postojeIzmene) {if(putanjaDatoteke == null) {putanjaDatoteke = getPath();}  snimiModel();}

	putanjaDatoteke = getPath();
	ucitajModel();

u snimiModel() se radi postojeIzmene = False
